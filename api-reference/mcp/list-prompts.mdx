---
title: "List MCP Prompts"
api: "GET /api/mcp/prompts"
description: "Retrieves all available prompts from a connected MCP server"
---

## Query Parameters

<ParamField query="id" type="string" required>
  MCP server ID
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Whether the request was successful
</ResponseField>

<ResponseField name="prompts" type="array">
  Array of available prompts
  <Expandable title="Prompt Object">
    <ResponseField name="name" type="string">
      Prompt name
    </ResponseField>
    <ResponseField name="description" type="string">
      Prompt description
    </ResponseField>
    <ResponseField name="arguments" type="array">
      Array of prompt arguments
      <Expandable title="Argument Object">
        <ResponseField name="name" type="string">
          Argument name
        </ResponseField>
        <ResponseField name="description" type="string">
          Argument description
        </ResponseField>
        <ResponseField name="required" type="boolean">
          Whether the argument is required
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>

```bash cURL
curl -X GET "https://api.example.com/api/mcp/prompts?id=mcp_abc123" \
  -H "Authorization: Bearer YOUR_API_KEY"
```

```javascript JavaScript
const response = await fetch(
  'https://api.example.com/api/mcp/prompts?id=mcp_abc123',
  {
    method: 'GET',
    headers: {
      'Authorization': 'Bearer YOUR_API_KEY'
    }
  }
);

const data = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://api.example.com/api/mcp/prompts',
    headers={'Authorization': 'Bearer YOUR_API_KEY'},
    params={'id': 'mcp_abc123'}
)

data = response.json()
```

</RequestExample>
