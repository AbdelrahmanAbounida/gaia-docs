---
title: "Get Credential"
api: "GET /api/credentials/{id}"
description: "Retrieves a specific credential by ID with masked API key"
---

## Path Parameters

<ParamField path="id" type="string" required>
  Credential ID
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="credential" type="object">
  The credential object
  
  <Expandable title="credential properties">
    <ResponseField name="id" type="string">
      Unique credential identifier
    </ResponseField>
    
    <ResponseField name="provider" type="string">
      Provider name (e.g., openai, anthropic, ollama)
    </ResponseField>
    
    <ResponseField name="baseUrl" type="string | null">
      Optional base URL for the provider
    </ResponseField>
    
    <ResponseField name="models" type="array">
      Array of available model names
    </ResponseField>
    
    <ResponseField name="isValid" type="boolean">
      Validation status of the credential
    </ResponseField>
    
    <ResponseField name="lastValidatedAt" type="string | null">
      ISO 8601 timestamp of last validation
    </ResponseField>
    
    <ResponseField name="createdAt" type="string">
      ISO 8601 timestamp of creation
    </ResponseField>
    
    <ResponseField name="maskedApiKey" type="string">
      Masked version of the API key (e.g., sk-...xyz)
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>

```bash cURL
curl --request GET \
  --url http://localhost:3000/api/v1/credentials/cred_123abc \
  --header 'Authorization: Bearer <token>'
```

```javascript JavaScript
const response = await fetch('http://localhost:3000/api/v1/credentials/cred_123abc', {
  method: 'GET',
  headers: {
    'Authorization': 'Bearer <token>'
  }
});

const data = await response.json();
```

```python Python
import requests

url = "http://localhost:3000/api/v1/credentials/cred_123abc"
headers = {"Authorization": "Bearer <token>"}

response = requests.get(url, headers=headers)
data = response.json()
```

</RequestExample>
